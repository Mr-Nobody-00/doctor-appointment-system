What: Maven Project Object Model — the build configuration file. It defines project metadata, Java version, and all library dependencies. Maven reads this file to download libraries and build the project.

CORS Configuration

What: Configures CORS (Cross-Origin Resource Sharing). Browsers block requests from one origin (localhost:5173) to another (localhost:8080) by default. This config tells Spring Boot to allow React to make API calls.

Without this: Every Axios call from React would fail with a CORS error.

import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@Configuration
public class WebConfig implements WebMvcConfigurer {
    @Override
    public void addCorsMappings(CorsRegistry registry) {
        registry.addMapping("/api/**")
                .allowedOrigins("http://localhost:5173")
                .allowedMethods("GET", "POST", "PUT", "DELETE");
    }
}

@Component

        this.userRepository = userRepository;

    @Override
    public void run(String... args) {

            User admin = new User();
            admin.setName("Admin");
            admin.setEmail("admin@demo.com");
            admin.setPassword("admin123");
            admin.setRole("ADMIN");
            userRepository.save(admin);
            System.out.println("Admin user created!");
        }
    }
}

Controller Public API

What: Handles user registration and login. These endpoints are public — no JWT token needed. On successful login, it returns a JWT token that the frontend stores and uses for all future requests.

AuthController.javaJava
@RestController
@RequestMapping("/api/auth")
public class AuthController {

    private final UserRepository userRepository;

    public AuthController(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    @PostMapping("/register")
    public User register(@RequestBody User user) {
        user.setRole("PATIENT");          // Always register as PATIENT
        return userRepository.save(user); // Save to database
    }

    @PostMapping("/login")
    public Map<String, String> login(@RequestBody Map<String, String> body) {
        String email = body.get("email");
        String password = body.get("password");

        User user = userRepository.findByEmail(email);
        if (user == null || !user.getPassword().equals(password)) {
            throw new RuntimeException("Invalid credentials");
        }

        String token = JwtUtil.generateToken(user.getEmail(), user.getRole());
        return Map.of("token", token, "role", user.getRole(), "name", user.getName());
    }
}

user.setRole("PATIENT")Force role to PATIENT — can't register as admin

AppointmentController.java — book()Java
@PostMapping  // POST /api/appointments
public Appointment book(@RequestBody Map<String, String> body) {
    // 1. Find the slot
    Slot slot = slotRepository.findById(Long.parseLong(body.get("slotId")))
        .orElseThrow();

    // 2. DOUBLE BOOKING CHECK - if already booked, reject!
    if (slot.getIsBooked()) {
        throw new RuntimeException("Slot already booked");
    }

    // 3. Lock the slot - mark as booked
    slot.setIsBooked(true);
    slotRepository.save(slot);

    // 4. Create the appointment
    User patient = userRepository.findByEmail(body.get("patientEmail"));
    Appointment appointment = new Appointment();
    appointment.setPatient(patient);
    appointment.setDoctor(slot.getDoctor());
    appointment.setSlot(slot);
    appointment.setStatus("CONFIRMED");
    appointment.setBookingDate(LocalDate.now().toString());

    return appointmentRepository.save(appointment);
}
AppointmentController.java — cancel()Java
@PutMapping("/{id}/cancel")  // PUT /api/appointments/5/cancel
public Appointment cancel(@PathVariable Long id) {

@GetMapping("/summary")  // GET /api/appointments/summary?date=2026-03-30
public Map<String, Object> getSummary(@RequestParam String date) {
    List<Appointment> all = appointmentRepository.findByBookingDate(date);

    // Group by mode: { "ONLINE": 5, "OFFLINE": 3 }
    Map<String, Long> byMode = all.stream()
        .collect(Collectors.groupingBy(
            a -> a.getDoctor().getMode(), Collectors.counting()));

    // Group by specialty: { "Cardiology": 4, "Dermatology": 2 }
    Map<String, Long> bySpecialty = all.stream()
        .collect(Collectors.groupingBy(
            a -> a.getDoctor().getSpecialty().getName(), Collectors.counting()));

    // Group by status: { "CONFIRMED": 6, "COMPLETED": 2 }
    Map<String, Long> byStatus = all.stream()
        .collect(Collectors.groupingBy(
            Appointment::getStatus, Collectors.counting()));

    return Map.of("total", all.size(),
        "byMode", byMode, "bySpecialty", bySpecialty, "byStatus", byStatus);
}

JwtUtil.java
src/main/java/com/example/demo/security/JwtUtil.java
Security

What: The token factory. Creates JWT tokens on login, extracts user info from tokens, and validates tokens. Uses HS256 signing algorithm with a secret key.

JwtUtil.javaJava
public class JwtUtil {

    // Secret key for signing tokens — generated once, stays in memory
    private static final Key SECRET_KEY = Keys.secretKeyFor(SignatureAlgorithm.HS256);
    private static final long EXPIRATION = 86400000; // 24 hours in milliseconds

    // CREATE a token from email + role
    public static String generateToken(String email, String role) {
        return Jwts.builder()
                .subject(email)              // who this token belongs to
                .claim("role", role)          // custom data: PATIENT or ADMIN
                .issuedAt(new Date())         // when token was created
                .expiration(new Date(         // when it expires (24h later)
                    System.currentTimeMillis() + EXPIRATION))
                .signWith(SECRET_KEY)         // sign with secret key
                .compact();                   // build the token string
    }

    // READ email from a token
    public static String extractEmail(String token) {
        return Jwts.parser()
                .setSigningKey(SECRET_KEY)   // use same key to verify
                .build()
                .parseSignedClaims(token)    // decode the token
                .getPayload()
                .getSubject();               // get the email (subject)
    }

    // CHECK if token is valid (not expired, not tampered)
    public static boolean isValid(String token) {
        try {
            extractEmail(token);  // if this works, token is valid
            return true;
        } catch (Exception e) {
            return false;         // expired, wrong signature, etc.
        }
    }
}

JwtFilter.java
src/main/java/com/example/demo/security/JwtFilter.java
Security

What: A security guard that intercepts EVERY HTTP request before it reaches any controller. It checks for a valid JWT token in the Authorization header. If the token is valid, the request is allowed through. If not, the request is blocked with 403 Forbidden.

OncePerRequestFilter: A Spring class that guarantees this filter runs exactly once per request.

JwtFilter.javaJava
public class JwtFilter extends OncePerRequestFilter {

    @Override
    protected void doFilterInternal(
            HttpServletRequest request,      // incoming HTTP request
            HttpServletResponse response,    // outgoing HTTP response
            FilterChain filterChain          // chain of other filters
    ) throws ServletException, IOException {

        // 1. Read the Authorization header
        String header = request.getHeader("Authorization");

        // 2. Check if it contains a Bearer token
        if (header != null && header.startsWith("Bearer ")) {
            String token = header.substring(7); // remove "Bearer " prefix

            // 3. Validate the token
            if (JwtUtil.isValid(token)) {
                String email = JwtUtil.extractEmail(token);

                // 4. Tell Spring Security "this user is authenticated"
                UsernamePasswordAuthenticationToken auth =
                    new UsernamePasswordAuthenticationToken(
                        email, null, Collections.emptyList());
                SecurityContextHolder.getContext().setAuthentication(auth);
            }
        }

        // 5. Continue to the next filter / controller
        filterChain.doFilter(request, response);
    }
}

SecurityConfig.java

Security

What: Defines the security rules — which URLs are public, which need authentication, and where our JWT filter sits in the chain.

SecurityConfig.javaJava
@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf(csrf -> csrf.disable())           // disable CSRF (not needed for REST APIs)
            .cors(cors -> cors.configure(http))      // enable CORS
            .sessionManagement(s -> s               // no sessions — we use JWT tokens
                .sessionCreationPolicy(SessionCreationPolicy.STATELESS))
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/auth/**").permitAll()   // login/register = PUBLIC
                .requestMatchers("/h2-console/**").permitAll() // DB console = PUBLIC
                .anyRequest().authenticated()                   // everything else = NEED TOKEN
            )
            .headers(headers -> headers             // allow H2 console in iframe
                .frameOptions(f -> f.disable()))
            .addFilterBefore(new JwtFilter(),        // add our JWT filter BEFORE
                UsernamePasswordAuthenticationFilter.class);  // Spring's default filter

        return http.build();
    }
}
